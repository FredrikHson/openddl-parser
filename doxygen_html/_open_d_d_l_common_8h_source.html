131  void set( const char *buffer, size_t numChars ) {
132  clear();
133  if( numChars > 0 ) {
134  m_len = numChars;
135  m_capacity = m_len + 1;
136  m_buffer = new char[ m_capacity ];
137  strncpy( m_buffer, buffer, numChars );
138  m_buffer[ numChars ] = '\0';
139  }
140  }

143  bool operator == ( const std::string &name ) const {
144  if( m_len != name.size() ) {
145  return false;
146  }
147  const int res( strncmp( m_buffer, name.c_str(), name.size() ) );
148 
149  return ( 0 == res );
150  }

153  bool operator == ( const Text &rhs ) const {
154  if( m_len != rhs.m_len ) {
155  return false;
156  }
157

159 
160  return ( 0 == res );
161  }
